JS API 예제

// 사용자 정보 주입
Feedback.identify({ email: 'user@example.com', name: 'Mike' });

// 이벤트 리스닝
Feedback.on('submitted', (payload) => {
  console.log('Issue created:', payload.issueId);
});

// 큐 패턴 (로드 전 호출 가능)
window.FeedbackQueue = [
  ['identify', { email: 'pre@example.com' }],
];
